#version 330 core

in vec3 position;
in vec2 textureCoords;

out vec2 blurTextureCoords[11];

uniform float targetHeight;

void main(void){
	gl_Position = vec4(position, 1.0);
	float pixelSize = 1.0 / targetHeight;
	
	for(int i = -5; i <= 5; i++) {
		blurTextureCoords[i + 5] = textureCoords + vec2(0.0, pixelSize * i);
	}
}
