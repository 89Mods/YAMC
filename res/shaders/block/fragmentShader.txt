#version 330 core

in vec2 pass_textureCoords;
in float pass_lightLevel;

in vec4 gl_Color;

out vec4 out_Color;
in float isSelected;

uniform sampler2D textureSampler;

void main(void){
	
	vec4 textureColor = texture(textureSampler, pass_textureCoords);
	if(textureColor.a < 0.5){
		discard;
	}
	float ll = pass_lightLevel;
	if(isSelected == 1.0) ll = 1.25;
	textureColor.r *= ll;
	textureColor.g *= ll;
	textureColor.b *= ll,
	
	out_Color = textureColor;
	
}
